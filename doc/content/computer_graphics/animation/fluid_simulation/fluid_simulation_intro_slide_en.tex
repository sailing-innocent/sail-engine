            \begin{equation}
                \begin{aligned}
                    \frac{\partial \vec{u}}{\partial t} + \vec{u}\cdot \nabla \vec{u} + \frac{1}{\rho}\nabla p & = \nu \nabla^2 \vec{u} + \vec{g} \\
                    \nabla \cdot \vec{u}                                                                       & = 0
                \end{aligned}
            \end{equation}
